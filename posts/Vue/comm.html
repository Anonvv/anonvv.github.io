1. 上下级组件通信 #

1.1. Props (上 -> 下) #

上级组件传参

<!-- index.vue -->
<script setup lang="ts">
import { Ref, ref } from 'vue'
import Sub from './Sub.vue'
const count: Ref<number> = ref(0)
</script>

<template>
    <div>
        <Sub :count="count" />
    </div>
</template>

1.2. emit (下 -> 上) #

上级组件接收

<!-- Sub.vue -->
<script setup lang="ts">
import Sub from './Sub.vue';

const subClick = (msg: string): void => {
    console.log('recieve: ', msg);
}
</script>
    
<template>
    <div>
        <Sub :count="count" @onClick="subClick" />
    </div>
</template>

下级组件传参

<script setup lang="ts">
import { Ref, ref } from 'vue'
const msg: Ref<string> = ref('msg from Sub')
const emits = defineEmits(['onClick'])
const onClick = (): void => {
    console.log('send:', msg.value);
    emits('onClick', msg.value)
}
</script>
        
<template>
    <div>
        <button @click="onClick">Send</button>
    </div>
</template>

2. 多级组件通信 #

2.1. provide / inject (上 -> 下) #

上级组件提供数据

<script setup lang="ts">
import { Ref, ref, provide, readonly } from "vue"
const msg: Ref<string> = ref('hello') 
provide('msg', readonly(msg)) // readonly 防止参数被下级组件修改
</script>

2.2. provide / inject (下 -> 上) #

上级组件想要接收跨层级子组件的数据，只要通过 provide 传递方法到下级组件，由下级组件调用，传回参数。

上级组件定义方法：

<script setup lang="ts">
import { provide } from "vue"
provide('func', (msg: string) => {
    console.log('msg from sub component:', msg)
})
</script>
