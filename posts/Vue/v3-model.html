Vue3 子组件更新父组件的参数

28 / 09 / 2022

父组件传参。

<!-- index.vue -->
<script setup lang="ts">
import { Ref, ref } from 'vue'
import Child from './Child.vue';
const count: Ref<number> = ref(0)
</script>
    
<template>
    <div class="index">
        {{ count }}
        <!-- 绑定参数 count -->
        <Child v-model:count="count" />
    </div>
</template>

子组件接收，并在事件中更新参数。

<!-- Child.vue -->
<script setup lang="ts">
// 定义接收参数
defineProps<{ count: number }>()
// 注册事件
const emits = defineEmits(['update:count'])
</script>
        
<template>
    <div class="child">
        <!-- 更新参数 update:propName -->
        <input type="text" @input="emits('update:count', $event.target.value)">
    </div>
</template>
