Vue 父组件访问子组件的数据或方法

03 / 08 / 2022

1. Vue2 #

父组件

<!-- index.vue -->
<template>
    <div>
        <!-- 1. 定义ref，该组件将会被父组件的 $refs 收集 -->
        <Child ref="childRef" />
    </div>
</template>
<script>
import Child from './Child.vue'
export default {
    mounted() {
        // 3. 页面加载完成，访问组件 ref
        //    - 如果该组件是 Vue 组件，返回组件实例
        //    - 如果该组件是普通标签，原样返回
        const childRef = this.$refs.childRef
        childRef.msg = 'world'
    }
}
</script>

2. Vue3 #

父组件

<!-- index.vue -->
<template>
    <div>
        <!-- 1. 定义ref -->
        <Child ref="childRef" />
    </div>
</template>
<script>
import { onMounted, ref } from 'vue'
import Child from './Child.vue'
// 2. 初始化
let childRef = ref(null)
// 3. 页面加载完成后，才能访问子组件暴露的属性
onMounted(() => {
    childRef.value.msg = 'world'
})
</script>

<!-- Child.vue -->
<script>
import { Ref, ref } from 'vue'
const msg: Ref<string> = 'hello'

// 通过 defineExpose 暴露的属性，外部才能访问
defineExpose({
    msg
})
</script>
