VitePress 实现本地搜索

20 / 09 / 2022

解析 markdown 文件内容。

1. 环境 #

  • VitePress(Vue3 + TS)

2. 思路 #

  1. 项目初始化时，生成索引表
    1. 创建一个空列表Array，存放文件解析对象；
    2. 递归遍历.md文件，通过插件grayMatter解析frontmatter信息，存入列表中；
    3. 遍历完成后，将Array转为json字符串，写入文件post.json，存入磁盘；
  2. 创建搜索组件，并监听input值改变
  3. 在组件中引入索引表，将文章标题title和内容content分别进行正则匹配，过滤出匹配成功的项目；
  4. 渲染匹配到的结果到搜索列表中，并设置关键字高亮。

3. 搜索模块代码 #

// ~/.vitepress/theme/util/search-map.js

import fs from 'fs'; // 文件模块
import path from 'path'; // 路径模块
import grayMatter from 'gray-matter' // 解析 frontmatter

// 索引列表
const fileMap = [];

/**
 * 获取目录文件索引
 * @param {string} targetDirPath 需要遍历的目录
 */
export default function createFilesJson(targetDirPath) {
    // 读取文件列表
    fs.readdirSync(targetDirPath).forEach(function (name) {
        // 获取文件完整路径
        const filePath = path.join(targetDirPath, name);
        // 获取文件信息
        const stat = fs.statSync(filePath);

        if (stat.isFile()) {
            // 保存文件索引对象
            fileMap.push(getFileObj(filePath))
        } else if ( // 过滤忽略文件夹
            stat.isDirectory() &&
            ".git"!=path.basename(filePath) &&
            '_' != path.basename(filePath).substr(0,1)
        ) {
            // 递归遍历目录
            createFilesJson(filePath);
        }
    });

    // 遍历完成，将 post.json 写入根目录
    fs.writeFileSync(path.resolve('./') + '/post.json', JSON.stringify(fileMap));
}

/**
 * 获取文件索引对象
 * @param {string} filePath 文件路径
 * @returns {object} 索引对象
 */
 function getFileObj(filePath) {
    if ( // 过滤忽略文件
        ".md" == path.extname(filePath).toLowerCase()
        && '_' != path.basename(filePath).substr(0,1) 
        && 'README.md' != path.basename(filePath)
        ){
        // 读取文件内容
        let dataStr = fs.readFileSync(filePath, 'utf8')
        // 解析 frontmatter
        const matterObj = grayMatter(dataStr)
        
        // 返回索引对象
        return {
            title: matterObj.data.title, // 标题
            content: matterObj.content, // 正文
            excerpt: matterObj.excerpt, // 摘要
            filePath: filePath // 文件路径
        }
    }
}

5. 创建搜索组件，监听关键字变化 #

样式可自行调整

<!-- search.vue -->

<template>
    <div>
        <!-- 搜索输入框 -->
        <input type="text" v-model="searchVal">
        <!-- 搜索结果列表 -->
        <ul>
            <li
                v-for="item in resultList"
                :key="item.title"
            >
                <a :href="item.filePath" @click="clearSearch">
                    <p v-html="item.title"></p>
                    <p v-html="item.content"></p>
                </a>
            </li>
            <!-- 搜索结果为空 -->
            <li v-if="!resultList.length && searchVal !== ''">Empty</li>
        </ul>
    </div>
</template>

<script setup lang="ts">
import { ref, reactive, watchEffect, computed } from "vue";
import postJson from '/post.json' // 引入生成的 post.json
// 关键字
const searchVal = ref('')
// 搜索结果列表
let resultList = reactive([])

// 监听输入
watchEffect(() => {
    localSearch(searchVal.value)
})

/**
 * 本地搜索
 * @param key 关键字
 */
function localSearch(key: string): void {
    resultList.length = 0
    if (key.trim() === '') { // 判空
        return
    }
    // 遍历索引表
    postJson.forEach(item => {
        // 对象深拷贝
        const newItem = JSON.parse(JSON.stringify(item))
        // 关键字正则
        const reg = new RegExp(key, 'ig')
        // 标题进行正则匹配，没有返回 null
        const titleMatches = newItem.title.match(reg)
        // 内容进行正则匹配，没有返回 null
        const contentMatches = newItem.content.match(reg)
        
        if (titleMatches || contentMatches) {
            // 匹配内容高亮
            newItem.content = highLight(key, newItem.content)
            newItem.title = highLight(key, newItem.title)
            // 保存到结果列表
            resultList.push(newItem)
        }
    })
}

/**
 * 关键字高亮
 * @param key 关键字
 * @param str 原内容
 * @return 新字符串
 */
const highLight = (key: string, str: string): string => {
    const reg = new RegExp(key, 'ig')
    return str.replace(reg, `<span style="color: red">${key}</span>`)
}

// 清空搜索
 const clearSearch = () => {
    searchVal.value = ''
}

</script>
