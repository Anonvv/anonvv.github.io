VitePress 生成文章目录

21 / 09 / 2022

1. 环境 #

  • Vue3
  • TypeScript

2. 思路 #

  1. 页面加载完成后，获取文章容器，查找容器内第一级的 h1-h6 标题元素节点列表；
  2. 创建一个空列表，遍历节点列表，获取节点信息(id、标签名称nodeName、标题内容innerTEXT、自定义下级列表)，封装为对象保存到创建的表中；通过childNodes查询元素下级元素，如果包含下级标题，继续递归，将结果保存到下级列表中；
  3. 拿到格式化后的数据，递归创建目录组件，每个控件需要包含a标签，将href绑定id，以实现锚点跳转。

3. 实现 #

3.1. 工具函数，获取文章标题树形列表 #

// util.ts

/**
 * 获取文章目录树形列表
 * @param parent 父级容器
 * @returns 标题树
 */
export const getCatalogTree = (parent: HTMLElement): Array<ICataItem> => {
    // 第一层标题
    const children: NodeList = parent.childNodes as NodeList
    return loop(children)
    /**
     * 递归遍历标题节点
     * @param nodes 标题节点集合
     * @returns 标题树
     */
    function loop(nodes: NodeList): Array<ICataItem> {
        if (!nodes || !nodes.length) return []
        // 标题树
        const titleList: Array<ICataItem> = []
        // 过滤出标题节点
        const hNodes: Node[] = Array.from(nodes).filter((item: Node) => {
            // 匹配h1-h6
            const reg = /^H+/ig;
            return reg.test(item.nodeName)
        })
        // 遍历标题节点
        for (let i = 0; i < hNodes.length; i++) {
            const item: Node = hNodes[i];
            titleList.push({
                nodeName: item.nodeName, 
                text: item.innerText,
                id: item.id,
                child: loop(item.childNodes) // 递归下级标题节点
            })
        }
        return titleList
    }
}

3.2. 递归创建视图组件 #

因为标签名未知，需要用函数式组件递归。

// catalog.ts

import { h } from 'vue'
// 创建目录虚拟节点
function createTitleNodes(tree: Array<ICataItem>): VNode[] {
    if (!tree.length) return []
    const list: VNode[] = []
    // 遍历目录树
    tree.forEach(item => {
        // 创建虚拟节点
        list.push(
            h(
                item.nodeName,
                { class: 'catalog-title' },
                h('a', { href: `#${item.id}`, name: item.id }, item.text.slice(0, item.text.length - 1))
            )
        )
        if (item.child?.length) {
            // 递归下级标题
            list.push(h(item.nodeName, createTitleNodes(item.child)))
        }
    })
    return list
}
const Catalog =  (props: { tree: Array<ICataItem> }) => {
    const tree = props.tree // 接收目录树
    return h('div', { class: 'catalog' }, createTitleNodes(tree))
}
export default Catalog

vue 单文件，直接引入页面使用

<!-- catalog.vue -->
<template>
    <Catalog v-if="isPost && cataData.length" :tree="cataData"></Catalog>
</template>

<script setup lang="ts">
import { onMounted, reactive, onUpdated, computed } from "vue";
import { useData } from 'vitepress'

import { getCatalogTree } from './util.ts'
import Catalog from './catalog.ts'

const { page } = useData()
const cataData: [] = reactive([]) // 目录树
const isPost: any = computed(() => page.value.relativePath.includes('posts'))

onMounted((): void => {
  init()
}),
// 初始化
const init = (): void => {
    if (isPost.value) {
        // 获取正文容器
        const contentDiv: ChildNode = document.getElementById('content')?.childNodes[0]
        // 获取目录
        cataData.push(...getCatalogTree(contentDiv))
    }
}
</script>
