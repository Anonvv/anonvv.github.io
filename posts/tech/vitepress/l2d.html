3. 创建 Live2D 组件 #

// ~/components/L2d/index.ts

import { h, render } from 'vue'

export default {
    render() {
        const vnode = h(
            'script',
            { src: 'https://cdn.jsdelivr.net/gh/stevenjoezhang/live2d-widget/autoload.js' }
        )
        render(vnode, document.body)
    },
}
