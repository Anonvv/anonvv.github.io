图片预加载

28 / 09 / 2022

预加载:
在页面加载完成前，提前加载图片等资源，有利于提高用户体验。

<script setup lang="ts">
import { onBeforeMount } from "vue";
// some image url
const urls: string[] = []
// before window loaded
onBeforeMount(() => {
    preloadImg()
})
// preload image
const preloadImg = (): void => {
    urls.forEach(url => {
        const img = new Image()
        img.src = url
        img.onload = () => {
            // loaded
        }
    })
}
</script>
