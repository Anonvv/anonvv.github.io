JavaScript 防抖节流

07 / 10 / 2022

性能优化

1. 防抖 #

在持续触发事件时，只执行一次事件处理函数。

1.1. 应用场景 #

常用于页面/鼠标滚动、提交按钮、输入框搜索等事件处理。

1.2. 实现思路 #

在第一次事件触发后，开始延时计时：

  • if: 在设定时间内再次触发了事件，取消上一次计时，开启新的延时计时；
  • else: 在设定时间内没有再次触发事件，执行事件处理函数。

1.3. 示例 #

/**
 * debounce
 * @param {Function} fn callback function
 * @param {Number} delay delay time 
 * @param  {...any} rest rest params 
 * @returns function
 */
const debounce = (fn, delay, ...rest) => { // rest 剩余参数列表
    let timer = null
    // 返回一个函数
    return (...args) => { // args 事件原始参数列表
        // 清除上一次定时
        if (timer) clearTimeout(timer)
        // 开启新的定时
        timer = setTimeout(() => {
            // 传回全部参数，并执行
            fn(...args, ...rest)
        }, delay);
    }
}

// 事件处理函数，延时1秒执行
const scrollFn = debounce((e, name) => {
    console.log('Hello,', name);
}, 1000, 'Tom')

// 监听滚动事件
window.addEventListener('scroll', scrollFn)

2. 节流 #

在持续触发事件时，在一定时间内只触发一次事件处理函数。

2.1. 应用场景 #

页面/鼠标滚动等需要稀释事件触发频率的场合。

2.2. 实现思路 #

在第一次事件触发后，开始延时计时：

  • 到达设定时间，触发一次事件，然后重置计时；
  • 如果在设定时间内再次触发了事件，忽略该次触发。

2.3. 示例 #

2.3.1. 定时器版 #

/**
 * throttle
 * @param {Function} fn callback function
 * @param {Number} delay delay time 
 * @param  {...any} rest rest params 
 * @returns function
 */
const throttle = (fn, delay, ...reset) => {
    let timer = null
    return (...args) => {
        // 计时器不存在，说明没有事件等待触发
        if (!timer) {
            // 设置定时器，延迟触发函数
            timer = setTimeout(() => {
                // 传回全部参数，并执行
                fn(...args, ...reset)
                // 函数触发完成后，重置定时器
                clearTimeout(timer)
                timer = null
            }, delay)
        }
    }
}

// 事件处理函数，1秒内只执行一次
const scrollFn = throttle((...args) => {
    console.log('Hell', args);
}, 1000, 'Tom')

// 监听滚动事件
window.addEventListener('scroll', scrollFn)

2.3.2. 时间戳版 #

/**
 * throttle
 * @param {Function} fn callback function
 * @param {Number} delay delay time 
 * @param  {...any} rest rest params 
 * @returns function
 */
const throttle = (fn, delay, ...reset) => {
    let startTime = new Date().getTime()
    return (...args) => {
        if (new Date().getTime() - startTime > delay) {
            fn(...args, ...reset)
            startTime = new Date().getTime()
        }
    }
}

// 事件处理函数，1秒内只执行一次
const scrollFn = throttle((...args) => {
    console.log('Hell', args);
}, 1000, 'Tom')

// 监听滚动事件
window.addEventListener('scroll', scrollFn)
