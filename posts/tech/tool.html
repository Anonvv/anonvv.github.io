  1. UpperCase or LowerCase a letter according to index by string replacement.
function fn(str: string, isUpperCase: boolean, index: number): string {
    if (isUpperCase) {
        return str.replace(str[index], str[index].toUpperCase())
    } else {
        return str.replace(str[index], str[index].toLowerCase())
    }
}
